🎯 Complete Direct Integration Template

Below is the minimal code structure that loads Paddle.js directly from the official CDN and sets up initialization prior to launching the checkout sheet:

example-checkout.html Paddle.js v2 SDK 
<!DOCTYPE html>
<html>
  <head>
    <!-- Always load directly from CDN -->
    <script src="https://cdn.paddle.com/paddle/v2/paddle.js"></script>
    
    <script type="text/javascript">
      // 1. Set sandbox environment *before* Initialize
      Paddle.Environment.set("sandbox");
      
      // 2. Initialize Paddle with Client-Side Token
      Paddle.Initialize({
        token: "test_99bce2255xxxxxxx" // Update token
      });

      // 3. Function to launch payment overlays
      function openCheckout() {
        Paddle.Checkout.open({
          items: [
            { priceId: "pri_01kqpe8ad9772rdsn3ddbw4bg3", quantity: 1 },
            { priceId: "pri_01kqpe9hv3r1v9wfxxvnjgq9zk", quantity: 1 },
            { priceId: "pri_01kqpebd19q7nppxkh53e0cnd3", quantity: 1 }
          ]
        });
      }
    </script>
  </head>
  <body>
    <button onclick="openCheckout()">Buy now</button>
  </body>
</html>
